namespace = c_take_claim


c_take_claim.01 = {
	type = country_event
	hidden = yes

	trigger = {
	}

	immediate = {
		
		random_scope_state = {
			limit = {
				has_variable = c_take_claim_owner
			}
			if = {
				limit = {
					NOT = {has_claim_by = var:c_take_claim_owner}
				}
				save_scope_as = target_state

				state_region = { add_claim = scope:target_state.var:c_take_claim_claimer }
				var:c_take_claim_owner = {
					remove_list_variable = {
						name = c_revoke_claim_owner
						target = scope:target_state
					}
				}
			}
			remove_variable = c_take_claim_owner
			remove_variable = c_take_claim_claimer
		}
		
	}
}

c_take_claim.02 = {
	type = country_event
	hidden = yes

	trigger = {
		has_variable_list = c_revoke_claim_claimer
		is_at_war = no
		is_diplomatic_play_committed_participant = no
	}

	immediate = {
		
		clear_variable_list = c_revoke_claim_claimer
		
	}
}
