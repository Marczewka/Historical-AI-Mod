on_wargoal_added = {
	on_actions = {
		take_claim_actions_1
	}
}

on_wargoal_enforced = {
	on_actions = {
		take_claim_actions_2
	}
}

#need this in case of white peace
on_monthly_pulse_country = {
	on_actions = {
		take_claim_actions_3
	}
}


take_claim_actions_1 = {
	effect = {
		if = {
			limit = {
				has_play_goal = revoke_claim
			}
			every_state = {
	    		limit = {
					owner = {
						is_country_type = decentralized
					}
	    			any_country = {
	    				is_diplomatic_play_enemy_of = scope:actor
	    				has_claim = prev
	    			}
	    		}
				save_scope_as = target_state

				scope:actor = {
	    			add_to_variable_list = {
	    				name = c_revoke_claim_claimer
                        target = scope:target_state
	    			}
	    		}

				root = {
					every_scope_play_involved = {
		            	limit = {
		            		NOT = { this = scope:actor}
		            		has_claim = scope:target_state
		            	}
		            	add_to_variable_list = {
		    				name = c_revoke_claim_owner
	                        target = scope:target_state
		    			}
		            }
				}
				
	    	}
		}
		
	}
}

take_claim_actions_2 = {
	effect = {
		if = {
			limit = {
				has_variable_list = c_revoke_claim_claimer
			}
			scope:target = {
				every_in_list = {
					variable = c_revoke_claim_owner
					root = { 
						remove_list_variable = {
							name = c_revoke_claim_claimer
							target = prev
						}
					}
					set_variable = {
						name = c_take_claim_claimer
						value = root
					}
					set_variable = {
						name = c_take_claim_owner
						value = scope:target
					}
					owner = { trigger_event = { id = c_take_claim.01 days = 1 } }

					
				}
			}
			
		}
	}

	
}


take_claim_actions_3 = {
	events = {
		c_take_claim.02
	}
}
