je_patagonia = {
	icon = "gfx/interface/icons/event_icons/event_military.dds"

	group = je_group_historical_content

	is_shown_when_inactive = {
		exists = c:PAT
			any_neighbouring_state = {
				OR = {
					owner = c:PAT
					owner = c:THL
				}
			}
		OR = {
			country_has_primary_culture = cu:south_andean
			country_has_primary_culture = cu:platinean
			country_has_primary_culture = cu:chilean
			country_has_primary_culture = cu:argentine
			country_has_primary_culture = cu:uruguayan
			country_has_primary_culture = cu:paraguayan
			country_has_primary_culture = cu:bolivian
			country_has_primary_culture = cu:peruvian
		}
	}

	scripted_button = patagonia_button

	possible = {
		NOT = { has_law = law_type:law_no_colonial_affairs }
	}

	complete = {
		OR = {
			owns_entire_state_region = STATE_ARAUCANIA
			owns_entire_state_region = STATE_PATAGONIA
		}
	}

	on_complete = {
		trigger_event = { id = patagonia.1 }
		hidden_effect = {
			set_variable = colonized_patagonia
		}
	}

	invalid = {
		s:STATE_ARAUCANIA = {
			OR = { 
				is_homeland = cu:south_andean
				is_homeland = cu:platinean
				is_homeland = cu:chilean
				is_homeland = cu:argentine
				is_homeland = cu:uruguayan
				is_homeland = cu:paraguayan
				is_homeland = cu:bolivian
				is_homeland = cu:peruvian
			}
		}
		s:STATE_PATAGONIA = {
			OR = { 
				is_homeland = cu:south_andean
				is_homeland = cu:platinean
				is_homeland = cu:chilean
				is_homeland = cu:argentine
				is_homeland = cu:uruguayan
				is_homeland = cu:paraguayan
				is_homeland = cu:bolivian
				is_homeland = cu:peruvian
			}
		}
	}

	on_yearly_pulse = {
		random_events = {
			40 = 0
			10 = patagonia.2
			7 = bp1_patagonia.1
		}
	}

	weight = 20
	should_be_pinned_by_default = yes
}
