gov_absolute_kingdom = {
	transfer_of_power = hereditary

	male_ruler = "RULER_TITLE_KING"
	female_ruler = "RULER_TITLE_QUEEN"
	
	male_heir = "RULER_TITLE_PRINCE"
	female_heir = "RULER_TITLE_PRINCESS"
	
	possible = {
		has_law_or_variant = law_type:law_monarchy
		has_law_or_variant = law_type:law_autocracy
		OR = {
			country_tier = kingdom
			c:ITA ?= this
			c:PRU ?= this
		}
		NAND = {
			is_subject_type = subject_type_personal_union
			any_overlord_or_above = {
				has_variable = cisleithania_var
			}
			OR = {
				has_variable = transleithania_var
				has_variable = triune_kingdom_var
			}
		}
		NOT = {
			is_subject_type = subject_type_crown_land
		}
	}

	on_government_type_change = {
		change_to_hereditary = yes
	}
	on_post_government_type_change = {
		post_change_to_hereditary = yes
	}
}
