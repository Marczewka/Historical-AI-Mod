christ_of_nations_button = {
	name = "christ_of_nations_button"
	desc = "christ_of_nations_button_desc"

	visible = { always = yes }

	possible = {
        is_subject = no
		country_rank >= rank_value:minor_power
		sr:region_poland = {
			any_scope_country = {
				NOT = { country_has_primary_culture = cu:polish }
				custom_tooltip = {
					text = owns_polish_homelands_tt
					any_scope_state = {
						state_region = {
							OR = {
								is_homeland = cu:polish
							}
						}
					}
				}
				relations:root <= 20
				any_political_movement = {
					is_political_movement_type = movement_cultural_minority
					culture ?= cu:polish
				}
				NOT = {
					any_political_movement = {
						has_modifier = modifier_polish_unificationist_sentiment
					}
				}
			}
		}
	}

	ai_chance = {
		value = 0
		if = {
			limit = {
				has_law = law_type:law_theocracy
			}
			add = -10
		}
		if = {
			limit = {
				any_country = {
					NOT = {
						country_has_primary_culture = cu:polish
					}
					any_scope_state = {
						state_region = {
							is_homeland = cu:polish
						}
					}
					relations:root <= 0
				}
			}
			add = 10
		}
		if = {
			limit = {
				any_country = {
					NOT = {
						country_has_primary_culture = cu:polish
					}
					any_scope_state = {
						state_region = {
							is_homeland = cu:polish
						}
					}
					relations:root <= -20
				}
			}
			add = 10
		}
		if = {
			limit = {
				is_at_war = yes
			}
			multiply = 0
		}
	}

	effect = {
		custom_tooltip = christ_of_nations_button
		ROOT = {
			save_scope_as = saboteur_country
		}
		every_country = {
			limit = {
				NOT = { country_has_primary_culture = cu:polish }
				any_scope_state = {
					state_region = {
						OR = {
							is_homeland = cu:polish
						}
					}
				}
				NOT = {
					any_political_movement = {
						is_political_movement_type = movement_cultural_minority
						OR = {
							culture = cu:polish
						}
						has_modifier = modifier_polish_unificationist_sentiment
					}
				}
			}
			every_scope_state = {
				limit = {
					state_region = { 
						is_homeland = cu:polish
					}
				}
				add_radicals_in_state = {
					value = 0.10
					culture = cu:polish
				}
				set_variable = {
					name = christ_of_nations_target
					days = long_modifier_time
				}
			}
			random_political_movement = {
				limit = {
					is_political_movement_type = movement_cultural_minority
					OR = {
						culture = cu:polish
					}
					NOT = { has_modifier = modifier_polish_unificationist_sentiment }
				}
				add_modifier = {
					name = modifier_polish_unificationist_sentiment
					days = short_modifier_time
				}
			}
			THIS = {
				save_scope_as = victim_country
			}
		}
	}
}

je_christ_of_nations_unification_button = {
	name = "je_christ_of_nations_unification_button"
	desc = "je_christ_of_nations_unification_button_desc"

	visible = { 
		NOT = {
			is_subject = yes
		}
	}

	cooldown = { days = 90 }

	possible = {
		has_technology_researched = nationalism
		is_at_war = no
		NOR = {
			is_revolutionary = yes
			is_secessionist = yes
		}
		custom_tooltip = {
			text = rejected_polish_union_pol_var_tt
			NOT = { has_variable = rejected_polish_union_pol_var }
		}
		any_country = {
			NOT = { this = root }
			is_at_war = no
			NOR = {
				is_revolutionary = yes
				is_secessionist = yes
			}
			country_has_primary_culture = cu:polish
			OR = {
				is_subject = no
				is_subject_of = ROOT
			}
			OR = {
				this.gdp < root.gdp
				root = { is_player = yes }
			}
			relations:root >= relations_threshold:amicable
			is_player = no
		}
	}

	ai_chance = {
		value = 100
	}

	effect = {
		trigger_event = { id = poland_events.3 popup = yes }
	}
}
